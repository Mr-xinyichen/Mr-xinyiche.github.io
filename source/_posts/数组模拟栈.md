```java
/**
 * @author cxy
 * 数组模拟栈
 */
public class ArrayStackDemo {
    public static void main(String[] args) {
        ArrayStack stack = new ArrayStack(4);
        stack.push(1);
        stack.push(2);
        stack.push(3);
        stack.push(4);
        stack.push(5);
        stack.pop();
        stack.push(6);
        stack.view();
    }
}

/**
 * 数组栈
 */
class ArrayStack{
    /**
     * 栈顶
     */
    private int top = -1;

    /**
     * 存放数据的数组
     */
    private int[] data;

    /**
     * 容量
     */
    private int maxSize;

    public ArrayStack(int maxSize) {
        this.data = new int[maxSize];
        this.maxSize = maxSize;
    }

    /**
     * 判满
     * @return
     */
    public boolean isFull(){
        return top == maxSize-1;
    }

    /**
     * 判空
     * @return
     */
    public boolean isEmpty(){
        return top == -1;
    }

    /**
     * 入栈
     * @param value
     */
    public void push(int value){
        if (isFull()){
            System.out.println("栈已满");
            return;
        }
        top++;
        data[top] = value;
    }

    /**
     * 弹栈 只是移动了栈顶指针 元素实际还在栈中
     * @param value
     */
    public int pop(){
        if (isEmpty()){
            throw new RuntimeException("栈中无元素");
        }
        int e = data[top];
        top--;
        return e;
    }

    /**
     * 查看栈中元素
     */
    public void view(){
        if (isEmpty()){
            System.out.println("栈中没有元素");
        }
        System.out.println("栈中一共有"+(top+1)+"个元素");
        for (int i = top; i > -1; i--) {
            System.out.println("索引"+i+"为"+data[i] );
        }
    }
}
```
