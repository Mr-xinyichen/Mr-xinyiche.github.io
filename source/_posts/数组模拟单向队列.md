/**
 * @author cxy
 * 数组模拟队列
 */
public class ArrayQueue<E> {

    /**
     * 默认容量
     */
    private final int initCapacity = 10;
    /**
     * 队列数组
     */
    private Object[] queue;
    /**
     * 头指针
     */
    private int head=0;
    /**
     * 尾指针
     */
    private int tail=0;
    /**
     * 元素个数
     */
    private int count=0;

    public ArrayQueue() {
        this.queue =  new Object[initCapacity];
        this.head = this.tail = this.count = 0;
    }

    public ArrayQueue(int capacity) {
        if (capacity<=0){
            capacity = this.initCapacity;
        }
        this.queue =  new Object[capacity];
        this.head = this.tail = this.count = 0;
    }

    /**
     * 判空
     * @return
     */
    public boolean isEmpty(){
        return count == 0;
    }

    /**
     * 判满
     * @return
     */
    public boolean isFull(){
        return count == queue.length;
    }

    /**
     * 入队
     * @param o
     * @return
     */
    public boolean addQueue(E o){
        //队列已满返回false
        if (isFull()){
            return false;
        }
        //当尾指针大于数组下标时要对插入的索引位置取余
        queue[tail % (queue.length)] = o;
        tail++;
        return true;
    }

    /**
     * 出队
     * @return
     */
    public E getQueue(){
        //队列为空返回null
        if (isEmpty()){
            return null;
        }
        head++;
        //当头指针大于数组下标时要对出队的索引位置取余
        return (E)queue[(head-1) % queue.length];
    }

    public static void main(String[] args) {
        ArrayQueue<Integer> queue = new ArrayQueue(2);
        System.out.println("add1");
        queue.addQueue(1);
        System.out.println("add2");
        queue.addQueue(2);
        Integer queue1 = queue.getQueue();
        System.out.println("get"+queue1);
        System.out.println("add3");
        queue.addQueue(3);
    }
}
